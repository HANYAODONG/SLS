## 4. 在 `model_cgta.py` 加入普通时间注意力

```python
import torch
import torch.nn as nn
import torch.nn.functional as F


class TemporalAttentionPooling(nn.Module):
    def __init__(self, feature_dim=1024, attention_dim=128, dropout=0.1):
        super().__init__()
        self.attention = nn.Sequential(
            nn.Linear(feature_dim, attention_dim),
            nn.Tanh(),
            nn.Dropout(dropout),
            nn.Linear(attention_dim, 1),
        )

    def forward(self, x):
        # x: [B, T, D]
        scores = self.attention(x)          # [B, T, 1]
        weights = torch.softmax(scores, 1)  # [B, T, 1]
        pooled = torch.sum(weights * x, 1)  # [B, D]
        return pooled, weights.squeeze(-1)
```

## 5. 在 `model_cgta.py` 加入通道引导时间注意力

```python
class ChannelGuidedTemporalAttention(nn.Module):
    def __init__(
        self,
        feature_dim=1024,
        attention_dim=128,
        dropout=0.1,
        use_std=True,
        use_stat_residual=True,
    ):
        super().__init__()

        self.use_std = use_std
        self.use_stat_residual = use_stat_residual
        statistic_dim = feature_dim * 2 if use_std else feature_dim

        self.channel_context_net = nn.Sequential(
            nn.Linear(statistic_dim, attention_dim),
            nn.SiLU(),
            nn.Dropout(dropout),
        )

        self.temporal_projection = nn.Linear(feature_dim, attention_dim)
        self.channel_projection = nn.Linear(attention_dim, attention_dim)
        self.score_projection = nn.Linear(attention_dim, 1)

        self.statistics_projection = nn.Sequential(
            nn.Linear(statistic_dim, attention_dim),
            nn.SiLU(),
            nn.Dropout(dropout),
            nn.Linear(attention_dim, feature_dim),
        )

        self.output_norm = nn.LayerNorm(feature_dim)

    def forward(self, x):
        # x: [B, T, D]
        mean = x.mean(dim=1)

        if self.use_std:
            std = torch.sqrt(
                torch.clamp(
                    torch.var(x, dim=1, unbiased=False),
                    min=1e-5,
                )
            )
            statistics = torch.cat([mean, std], dim=-1)
        else:
            statistics = mean

        channel_context = self.channel_context_net(statistics)       # [B, A]
        temporal_feature = self.temporal_projection(x)               # [B, T, A]
        channel_feature = self.channel_projection(channel_context)    # [B, A]
        channel_feature = channel_feature.unsqueeze(1)                # [B, 1, A]

        joint_feature = torch.tanh(temporal_feature + channel_feature)
        scores = self.score_projection(joint_feature)                 # [B, T, 1]
        temporal_weights = torch.softmax(scores, dim=1)
        temporal_pooled = torch.sum(temporal_weights * x, dim=1)       # [B, D]

        if self.use_stat_residual:
            statistics_feature = self.statistics_projection(statistics)
            pooled = self.output_norm(temporal_pooled + statistics_feature)
        else:
            pooled = self.output_norm(temporal_pooled)

        return pooled, temporal_weights.squeeze(-1), channel_context
```

## 6. 在 `Model.__init__()` 中初始化

```python
self.use_swiglu = bool(int(getattr(args, "use_swiglu", 1)))
self.pooling_type = getattr(args, "pooling_type", "cgta")
self.cgta_use_std = bool(int(getattr(args, "cgta_use_std", 1)))

    int(getattr(args, "cgta_stat_residual", 1))
)

self.temporal_attention = TemporalAttentionPooling(
    feature_dim=1024,
    attention_dim=128,
    dropout=0.1,
)

self.cgta_pooling = ChannelGuidedTemporalAttention(
    feature_dim=1024,
    attention_dim=128,
    dropout=0.1,
    use_std=self.cgta_use_std,
    use_stat_residual=self.cgta_stat_residual,
)

self.cgta_classifier = nn.Sequential(
    nn.LayerNorm(1024),
    nn.Linear(1024, 256),
    nn.SELU(inplace=True),
    nn.Dropout(0.2),
    nn.Linear(256, 2),
)

self.last_layer_weights = None
self.last_temporal_weights = None
self.last_channel_context = None
```

如果 XLS-R 输出维度不是 `1024`，将以上所有 `1024` 改成真实的 `D`。

## 7. 修改 `forward()`

在已经获得：

```python
hidden_states  # [B, L, T, D]
```

之后加入：

```python
fused, layer_weights = self.stat_sls(hidden_states)
# fused: [B, T, D]

if self.use_swiglu:
    fused = self.swiglu(fused)

channel_context = None

if self.pooling_type == "max":
    pooled = torch.max(fused, dim=1)[0]
    temporal_weights = None

elif self.pooling_type == "mean":
    pooled = torch.mean(fused, dim=1)
    temporal_weights = None

elif self.pooling_type == "attention":
    pooled, temporal_weights = self.temporal_attention(fused)

elif self.pooling_type == "cgta":
    pooled, temporal_weights, channel_context = self.cgta_pooling(fused)

else:
    raise ValueError("Invalid pooling_type: " + str(self.pooling_type))

logits = self.cgta_classifier(pooled)
output = F.log_softmax(logits, dim=1)

self.last_layer_weights = layer_weights.detach()
self.last_temporal_weights = (
    temporal_weights.detach() if temporal_weights is not None else None
)
self.last_channel_context = (
    channel_context.detach() if channel_context is not None else None
)

return output
```

必须保持原 `main.py` 期待的返回值数量。若原项目要求：

```python
return output, feature
```

则改为：

```python
return output, pooled
```

## 9. 新建模块测试 `test_cgta_modules.py`

```python
import torch

from model_cgta import (
    TemporalAttentionPooling,
    ChannelGuidedTemporalAttention,
)


def main():
    batch_size = 2
    time_steps = 201
    feature_dim = 1024

    x = torch.randn(batch_size, time_steps, feature_dim)

    attention = TemporalAttentionPooling(
        feature_dim=feature_dim,
        attention_dim=128,
    )
    pooled, weights = attention(x)

    assert pooled.shape == (batch_size, feature_dim)
    assert weights.shape == (batch_size, time_steps)
    assert torch.allclose(
        weights.sum(dim=1),
        torch.ones(batch_size),
        atol=1e-5,
    )

    cgta = ChannelGuidedTemporalAttention(
        feature_dim=feature_dim,
        attention_dim=128,
        use_std=True,
        use_stat_residual=True,
    )

    pooled, weights, context = cgta(x)

    assert pooled.shape == (batch_size, feature_dim)
    assert weights.shape == (batch_size, time_steps)
    assert context.shape == (batch_size, 128)
    assert torch.allclose(
        weights.sum(dim=1),
        torch.ones(batch_size),
        atol=1e-5,
    )

    pooled.mean().backward()
    print("All CGTA module tests passed.")


if __name__ == "__main__":
    main()
```
